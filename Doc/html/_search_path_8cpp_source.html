106 void SearchPath::AnalysisPath( void )
107 {
108  m_path_toks.Clear();
109 
110  Util::String str_copy = m_path;
111  //str_copy.ClearAllWhiteChars();
112  //str_copy.ConvertToLowercast();
113 
114  int i = 0;
115  Token curr_token;
116  curr_token.Type( Token::PATH_NODE_TOKEN );
117  while( i < str_copy.Length() )
118  {
119  Util::String::XCHAR c = str_copy.At( i );
120  //若不是目录分隔符
121  if( c != '\\' && c != '"' && c != '“' )
122  {
123  curr_token.Append( c );
124  }else{
125 
126  //若当前符号不为空则加入列表
127  if( !curr_token.IsEmpty() )
128  {
129  curr_token.Analysis();
130  m_path_toks.PushBack( curr_token );
131  }
132  curr_token.Clear();
133  curr_token.Type( Token::PATH_NODE_TOKEN );
134  }
135  i++;
136  }//while( i < str_copy.Length() )
137 
138  //若当前符号不为空则放入列表
139  if( !curr_token.IsEmpty() )
140  {
141  curr_token.Analysis();
142  m_path_toks.PushBack( curr_token );
143  curr_token.Clear();
144  }
145 }
