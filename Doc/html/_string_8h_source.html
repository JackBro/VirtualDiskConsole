276  {
277  //若两个字符串都为空 则判定两个字符串相等
278  if( this->IsEmpty() && str.IsEmpty() )

294  T c1 = *p;
295  T c2 = *q;
296

299  { //若字符为大写则转换为小写再进行比较

306  c2 += 0x20;

308  }

311  {

314

316  q++;
317  }
318

320  {
321  return -1;
322  }else if( ret > 0 ){
323  return 1;
